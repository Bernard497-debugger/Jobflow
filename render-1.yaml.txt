services:
  # PostgreSQL Database
  - type: pserv
    name: jobflow-db
    ipAllowList: []
    postgresVersion: 15
    plan: free

  # Flask Web Service
  - type: web
    name: jobflow
    env: python
    pythonVersion: 3.11.0
    plan: free
    buildCommand: pip install -r requirements_production.txt
    startCommand: gunicorn app_production:app
    healthCheckPath: /
    healthCheckStartupFailureThreshold: 10
    
    # Environment Variables
    envVars:
      - key: DATABASE_URL
        fromDatabase:
          name: jobflow-db
          property: connectionString
      - key: OPENROUTER_API_KEY
        sync: false
      - key: APP_ID
        sync: false
      - key: APP_KEY
        sync: false
      - key: SECRET_KEY
        sync: false
      - key: FLASK_ENV
        value: production
      - key: APP_URL
        fromService:
          name: jobflow
          property: url
      - key: PORT
        value: "5000"
    
    # Auto-deploy on GitHub push
    autoDeploy: true
    
    # Region
    region: onrender
    
    # Disk (optional, for logs)
    disk:
      name: jobflow-logs
      sizeGB: 1
      mountPath: /app/logs
